303 void Link(list_t* list, list_link_t* link) {
304  // if the list is empty then put this item on the front
305  if(list->first == 0) {
306  BlockInterrupts();
307  list->first = link;
308  list->last = link;
309  link->next = NULL_LINK;
310  link->previous = NULL_LINK;
311  list->length++;
312  RestoreInterrupts();
313  return;
314  }
315  if(list->sort_fn) {
316  list_link_t* current;
317  current = list->first;
318  // see if this task will be linked into the first position
319  if(list->sort_fn(link->item_ptr, current->item_ptr)) {
320  BlockInterrupts();
321  list->first->previous = link;
322  link->next = list->first;
323  link->previous = NULL_LINK;
324  list->first = link;
325  list->length++;
326  RestoreInterrupts();
327  return;
328  }
329  while(current) {
330  if(list->sort_fn(link->item_ptr, current->item_ptr)) {
331  BlockInterrupts();
332  link->previous = current->previous;
333  link->next = current;
334  current->previous->next = link;
335  current->previous = link;
336  list->length++;
337  RestoreInterrupts();
338  return;
339  }
340  current = current->next;
341  }
342  }
343  // if there is no sort then link the item to the end (last) of the list
344  // same if we traversed the entire list and didn't find where it went yet
345  BlockInterrupts();
346  link->previous = list->last;
347  if(list->last) {
348  list->last->next = link;
349  }else list->first = link;
350  list->last = link;
351  link->next = NULL_LINK;
352  list->length++;
353  RestoreInterrupts();
354 }

373  RestoreInterrupts();
374  return;
375  }

377  list->last = link->previous;
378  if(list->last == NULL_LINK) {
379  list->first = NULL_LINK;
380  list->length = 0;
381  }else {

391  link->next = LINK_IN_USE;
392  list->length--;
393  RestoreInterrupts();
