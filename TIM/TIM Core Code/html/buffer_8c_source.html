19 void Push(buffer_t *buffer, char data) {
20  // disable interrupts to make the function interrupt safe
21  BlockInterrupts();
22 
23  /* If front = rear and the buffer is not empty then overwrite what is on
24  the front, this allows use as a revolving circular buffer for receivers */
25  if(buffer->front == buffer->rear && buffer->size) {
26  buffer->front++;
27  // check to make sure front isn't past the end of the buffer
28  if(buffer->front > buffer->buffer_end)
29  buffer->front = buffer->buffer_start;
30  buffer->size--;
31  }
32 
33  // push data onto rear location and increment rear pointer
34  *buffer->rear++ = data;
35  // check to make sure rear isn't past the end of the buffer
36  if(buffer->rear > buffer->buffer_end) buffer->rear = buffer->buffer_start;
37  // Increment the size of the buffer
38  buffer->size++;
39 
40  // If there is a callback, call it
41  if(buffer->Callback) buffer->Callback(buffer);
42 
43  //ReEnable Interrupts

45 }
