123 void ReceiveChar(char c) {
124  //static char receive_buffer[LOG_RECEIVE_MAX_LENGTH];
125  static enum com_state receive_state = IDLE;
126  static int length;
127  if(sys.settings.echo) {
128  UART_WriteByte(SUBSYS_UART, c);
129  if(c == RECEIVE_STOP_CHAR) UART_WriteByte(SUBSYS_UART, '\n');
130  }
131  if (receive_state == IDLE) {
132  if (c == RECEIVE_START_CHAR) {
133  receive_state = WRITING;
134  length = 0;
135  }
136  } else {
137  receive_buffer[length] = c;
138  if (c == RECEIVE_STOP_CHAR) {
139  // process received message
140  ProcessMsg();
141  receive_state = IDLE;
142  } else {
143  if (c == ASCII_BACKSPACE) {
144  if (length) length--;
145  else receive_state = IDLE;
146  } else {
147  length++;
148  if (length >= RECEIVE_MAX_LENGTH) receive_state = IDLE;
149  }
150  }
151  }
152 }

154 void ProcessMsg(void) {
155  volatile int16_t i = 0;
156  char *name;
157  char *argv[RECEIVE_MAX_ARGC];
158  int argc = 0;
159  // get the name
160  name = &receive_buffer[0];
161 
162  while (receive_buffer[i] != RECEIVE_STOP_CHAR) {
163  // if the index is greater than length then break;
164  if (i >= RECEIVE_MAX_LENGTH) break;
165  if (receive_buffer[i] == ' ') {
166  receive_buffer[i] = 0;
167  break;
168  }
169  i++;
170  }
171  // receive the arguments
172  argv[argc] = &receive_buffer[++i];
173  while (receive_buffer[i] != RECEIVE_STOP_CHAR) {
174  // if the index is greater than length then break;
175  if (i >= RECEIVE_MAX_LENGTH) break;
176  i++;
177  if (receive_buffer[i] == ' ' || receive_buffer[i] == RECEIVE_STOP_CHAR) {
178  argc++;
179  if(receive_buffer[i] == RECEIVE_STOP_CHAR) {
180  receive_buffer[i] = 0;
181  break;
182  }else {
183  receive_buffer[i] = 0;
184  argv[argc] = &receive_buffer[++i];
185  }
186  }
187  }
188 
189  for (i = 0; i <= sys.last_subsys; i++) {
190  if (strcasecmp(sys.subsystem[i].name, name) == 0) {
191  if (sys.subsystem[i].callback) {
192  sys.subsystem[i].callback(argc, argv);
193  }
194  }
195  }
196 }

299 static void SystemCallback(int argc, char *argv[]) {
300  volatile uint8_t i;
301  if(argc) {
302  if(strcasecmp(argv[0],"list") == 0) {
303  for(i = 0; i < MAX_SUBSYSTEMS; i++) {
304  if(sys.subsystem[i].name) LogStr("%d: %s\r\n", i, sys.subsystem[i].name);
305  }
306  }else if(strcasecmp(argv[0],"mute") == 0) {
307  LogMsg(sys.id, "muted");
308  sys.settings.mute = 1;
309  }else if(strcasecmp(argv[0],"unmute") == 0) {
310  sys.settings.mute = 0;
311  if(argc >= 2 && strcasecmp(argv[1], "all") == 0) {
312  for(i = 0; i <= sys.last_subsys; i++) sys.subsystem[i].log = 0;
313  LogMsg(sys.id, "unmuted all");
314  }else LogMsg(sys.id, "unmuted");
315  }else if(strcasecmp(argv[0],"echo") == 0) {
316  if(argc >= 2 && strcasecmp(argv[1],"off") == 0) {
317  sys.settings.echo = 0;
318  LogMsg(sys.id, "echo off");
319  }else {
320  sys.settings.echo = 1;
321  LogMsg(sys.id, "echo on");
322  }
323  }else{
324  //check if the command is actually a subsystem name
325  for(i = 0; i <= sys.last_subsys; i++) {
326  if(strcasecmp(argv[0], sys.subsystem[i].name) == 0) {
327  if(argc < 2) LogMsg(sys.id, "to few args");
328  else {
329  if(strcasecmp(argv[1],"mute") == 0) {
330  LogMsg(sys.id, "muted");
331  sys.subsystem[i].log = 0;
332  }else if(strcasecmp(argv[1],"unmute") == 0) {
333  sys.subsystem[i].log = 1;
334  LogMsg(sys.id, "unmuted");
335  }else LogMsg(sys.id, "cmd not supported");
336  }
337  return;
338  }
339  }
340  LogMsg(sys.id, "cmd not supported");
341  }
342  }
