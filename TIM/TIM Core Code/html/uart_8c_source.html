338 void UART_Tick(void){
339  char c;
340  volatile unsigned int i;
341  volatile unsigned int b;
342 
343  if(rx_flags != 0){
344  for(i = 0; i < NUM_UARTS; i++){
345  if(rx_flags & (1 << i)){
346  // clear the flag
347  rx_flags &= ~(1 << i);
348  b = GetSize(uart[i].rx);
349  if(b > UART_MAX_CHARS_PER_CALL) b = UART_MAX_CHARS_PER_CALL;
350  while(b--){
351  c = Pop(uart[i].rx);
352  CharReceiverList_Run(uart[i].receiverList, c);
353  }
354  }
355  }
356  }
357 }
